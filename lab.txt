
struct Node {
    int data;
    Node* next;
};

int count(Node* head, int val) {
    int cnt = 0;
    Node* curr = head;

    // Traverse the list
    while (curr != nullptr) {
        if (curr->data == val) {
            cnt++;
        }
        curr = curr->next;
    }

    return cnt;
}
